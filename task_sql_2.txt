Составить запрос, который выдаст список фамилий клиентов, записанных на просмотр 2х и более 3х комнатных квартир 

SELECT last_name
FROM client
INNER JOIN view ON client.id=view.client_id
INNER JOIN appartment ON view.appartment_id=appartment.id
WHERE appartment.rooms>3
 GROUP BY last_name
HAVING COUNT(view.client_id)>=2;
